【CSP】回收站选址

2023/3/6 算法题 csp 排序 模拟

该题目由于设定了坐标x和y的最大值为1e9，因此不能直接将坐标作为数组的下标。

考虑到n最大为1000，O(n^2)的算法足够解决问题，因此用结构体数组存储每个坐标，然后对于数组进行排序（按x的升序，如果x坐标相同则按y坐标的升序）。

从样例1中的图来看，我们可以遍历每一列（同一列中x坐标相同），将该列与相邻的两列进行比较，从而得到那些可以设定垃圾站的点，为方便遍历可以用如下所示的结构体进行处理：

struct COL{
    ll begin;  // 每一列的第一个数据在输入数组中的坐标
    ll len;    // 列的长度
};

在程序的主体部分，我们只遍历x坐标相邻的三列数据，然后再对于每列数据进行分别遍历。

以样例1为例，遍历中间一列，从第2个数据开始，即（1,1），先检查（1,1）上下两个坐标点是在其附近；然后，对于左右两列，从第一个数据遍历，找到y坐标等于（1，1）点的坐标，如果能找到则满足条件。最后，对于满足条件的点，检查其四顶角是否有垃圾。当然，这些遍历的实现是建立在我们已经排好序的情况下。

具体代码如下：

#include <iostream>
#include <algorithm>

using namespace std;

typedef long long int ll;

ll n;
struct POS
{
    ll x;
    ll y;
};

POS pos[1005];

struct COL{
    ll begin;
    ll len;
};

COL col[1005];
ll nCol = 1;

bool cmp(const POS &a, const POS &b)
{
    if (a.x != b.x)
        return a.x < b.x;
    else
        return a.y < b.y;
}
ll ans[5] = {0};
int main()
{
    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        cin >> pos[i].x >> pos[i].y;
    }
    sort(pos + 1, pos + 1 + n,  cmp);
    col[nCol].begin = 1;
    for (int i = 2; ; )
    {
        while (i <= n && pos[i].x == pos[i - 1].x)
        {
            i++;
        }
        col[nCol].len = i - col[nCol].begin;
        if (i > n) break;
        nCol++;
        col[nCol].begin = i++;
    }
    if (nCol < 3)
    {
        for (int i = 1; i <= 5; i++)
            cout << 0 << endl;
        system("pause");
        return 0;
    }
    // i1, i2, i3是针对每一列而言的偏移量，需要将其转换到pos中的实际下标
    for (int i = 1; i + 2 <= nCol; i++)
    {
        // 如果列不相邻，直接跳过
        if (pos[col[i].begin].x + 1 != pos[col[i + 1].begin].x || pos[col[i + 1].begin].x + 1 != pos[col[i + 2].begin].x) continue;
        int i1, i3;
        i1 = i3 = 1;
        for (int i2 = 2; i2 <= col[i + 1].len - 1 && i1 <= col[i].len && i3 <= col[i + 2].len; i2++)
        {
            int j2 = i2 + col[i + 1].begin - 1;
            // 上下存在垃圾
            if (pos[j2 - 1].y + 1 != pos[j2].y || pos[j2].y + 1 != pos[j2 + 1].y) continue;
            while (i1 <= col[i].len && pos[i1 + col[i].begin - 1].y < pos[j2].y)
            {
                i1++;
            }
            // 左存在垃圾
            if (i1 <= col[i].len && pos[i1 + col[i].begin - 1].y == pos[j2].y)
            {
                while (i3 <= col[i + 2].len && pos[i3 + col[i + 2].begin - 1].y < pos[j2].y)
                {
                    i3++;
                }
                // 右存在垃圾
                if (i3 <= col[i + 2].len && pos[i3 + col[i + 2].begin - 1].y == pos[j2].y)
                {
                    // 检查四个角落是否有垃圾
                    int tmpAns = 0;
                    int j1 = i1 + col[i].begin - 1;
                    int j3 = i3 + col[i + 2].begin - 1;
                    if (pos[j1].y == pos[j1 - 1].y + 1) tmpAns++;
                    if (pos[j3].y == pos[j3 - 1].y + 1) tmpAns++;
                    if (i1 + 1 <= col[i].len && pos[j1].y + 1 == pos[j1 + 1].y) tmpAns++;
                    if (i3 + 1 <= col[i + 2].len && pos[j3].y + 1 == pos[j3 + 1].y) tmpAns++;
                    ans[tmpAns]++; 
                }     
            }

        }
    }
    for (int i = 0; i < 5; i++)
        cout << ans[i] << endl;

    system("pause");
}
