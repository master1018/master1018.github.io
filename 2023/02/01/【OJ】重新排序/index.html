【OJ】重新排序

2023/2/1 算法题 前缀和 蓝桥杯 排序

重新排序 - 蓝桥云课 (lanqiao.cn)

该题的关键点：如何获得重新排序的查询区间之和的最大值。

可以这样进行考虑，我们希望数组中最大的几个数在我们查询的区间中，这样一来，所查询的和就是数组中最大的几个数的和。另外，题目中区间是可以有重叠的，因此我们可以首先统计每个下标的查询次数，然后将数组进行排序，查询次数最大的坐标存放数组中最大的数（以此类推），这样我们求得的和就是最大。

统计次数用了前缀和和差分进行处理（不知道正常的遍历会不会超时。

AC代码：

#include <iostream>
#include <algorithm>
using namespace std;

typedef unsigned long long int ll;

ll sum1[100005], sum2[100005];
ll arr[100005];
long long int query[1000005];

bool cmp(const int a, const int b)
{
    return a >  b;
}

int main()
{
    ll n;
    cin >> n;
    for (int i = 1; i <= n; i++)
        cin >> arr[i];
    for (int i = 1; i <= n; i++)
        sum1[i] += sum1[i - 1] + arr[i];
    ll m;
    cin >> m;
    ll len = 0;
    ll query_sum = 0;
    for (int i = 1; i <= m; i++)
    {
        ll L, R;
        cin >> L >> R;
        // 查询的区间内，坐标的访问次数 +1
        query[L] += 1, query[R + 1] -= 1; 
        query_sum += sum1[R] - sum1[L - 1];
    }
    for (int i = 1; i <= n + 1; i++)
        query[i] += query[i - 1];
    sort(arr + 1, arr + 1 + n, cmp);
    sort(query + 1, query + 1 + n, cmp);
    ll query_max = 0;
    for (int i = 1; query[i] != 0; i++)
    {
        query_max += arr[i] * query[i];
    }
    cout << query_max - query_sum << endl;

    system("pause");
}
