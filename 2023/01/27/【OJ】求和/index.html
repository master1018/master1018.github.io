【OJ】求和

2023/1/27 算法题 前缀和

题目来源：P2671 [NOIP2015 普及组] 求和 - 洛谷 | 计算机科学教育新生态 (luogu.com.cn)

题目中给定三元组（x，y，z），可以发现y根本没有什么作用，只要（x + z）/ 2是一个整数就行了。那么我们所求的三元组就是同奇偶的x和z。

在计算之前，先进性排序处理，首先将相同color的数放在一起，然后对于相同color的数据，将其按照奇偶进行分开，因此需要自定义cmp函数：

bool cmp(const Node &a, const Node &b)
{
    if (a.col != b.col)
        return a.col < b.col;
    else
        return a.idx % 2 < b.idx % 2;
}

排好序后，就可以用双指针，遍历x和z，将符合条件的数进行计算，但这样做最终只能过8个检测点（O2优化可过9个），因此要想全部AC需要继续优化算法，很显然，计算方式是优化的一个地方。

考虑以下式子：

n = 1
ans = f1 * a1

n = 2
ans = (f1 + f2) * (a1 + a2)

n = 3
ans = (f1 + f2) * (a1 + a2) + (f1 + f3) * (a1 + a3) + (f2 + f3) * (a2 + a3)
    = (f1 + f2 + f3) * (a1 + a2 + a3) + f1 * a1 + f2 * a2 + f3 * f3

n = 4
ans = (f1 + f2 + f3 + f4) * (a1 + a2 + a3 + a4) + 2 * (f1 * a1 + f2 * a2 + f3 * a3 + f4 * a4)

于是可以得到计算的通式：

ans(n) = sum_f(n) * sum_a(n) + (n - 2) * sum_fa(n)

因此我们可以转换为前缀和的形式来进行计算，可以大大降低时间复杂度，代码如下：

//#pragma GCC optimize(2)
#include <iostream>
#include <algorithm>
using namespace std;

typedef unsigned long long int ll;


#define N (10007)
ll n, m;

struct Node{
    ll idx;
    ll num;
    ll col;
};

ll sumf[100005], suma[100005], sumfa[100005];

Node arr[100005];

int read() {
    int x=0,f=1;
    char c=getchar();
    while(c<'0'||c>'9'){if(c=='-') f=-1;c=getchar();}
    while(c>='0'&&c<='9') x=x*10+c-'0',c=getchar();
    return x*f;
}

void write(int x) {
     if(x<0) putchar('-'),x=-x;
     if(x>9) write(x/10);
     putchar(x%10+'0');
}

bool cmp(const Node &a, const Node &b)
{
    if (a.col != b.col)
        return a.col < b.col;
    else
        return a.idx % 2 < b.idx % 2;
}

int main()
{
    cin >> n >> m;
    for (ll i = 1; i <= n; i++)
        arr[i].num = read();
    for (ll i = 1; i <= n; i++)
        arr[i].col = read();
    for (ll i = 1; i <= n; i++)
        arr[i].idx = i;
    sort(arr + 1, arr + n + 1, cmp);
    sumf[0] = suma[0] = sumfa[0] = 0;
    sumf[1] = arr[1].num, suma[1] = arr[1].idx, sumfa[1] = arr[1].idx * arr[1].num; 
    for (int i = 2; i <= n; i++)
        sumf[i] = sumf[i - 1] + arr[i].num, suma[i] = suma[i - 1] + arr[i].idx, sumfa[i] += sumfa[i - 1] + arr[i].idx * arr[i].num;  
    ll ans = 0;
    //for (int i = 1; i <= n; i++)
      //  cout << arr[i].idx << " " << arr[i].col << " " << arr[i].num << endl;
    for (ll i = 1; i <= n; )
    {
        int j = i + 1;
        while (j <= n && arr[j].col == arr[i].col && !((arr[j].idx + arr[i].idx) % 2)) j++;
        j--;
        if (j > i)
            ans += (sumf[j] - sumf[i - 1]) * (suma[j] - suma[i - 1]) + (j - i - 1) * (sumfa[j] - sumfa[i - 1]) % N;
        i = j + 1;
    }
    cout << ans % N << endl;
    system("pause");
}
